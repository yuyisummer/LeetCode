class Solution {
    
    int res = 0;
    
    public void helper(int[] nums, int S, int p){
        if(p >= nums.length)
        {
            if(S == 0) res++;
            return ;
        }
        helper(nums, S - nums[p], p + 1);
        helper(nums, S + nums[p], p + 1);
    }

    public int findTargetSumWays(int[] nums, int target) {
        helper(nums, target, 0);  //é€’å½’
        return res;
    }
}
